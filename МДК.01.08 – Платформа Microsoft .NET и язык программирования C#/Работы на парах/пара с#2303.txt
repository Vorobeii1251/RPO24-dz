using System;

public static class CurrencyConverter
{
    private static double kurs = 75.5;

    public static void SmenitKurs(double noviyKurs)
    {
        kurs = noviyKurs;
        Console.WriteLine($"Новый курс: 1$ = {kurs} руб.");
    }

    public static double DollaryVRubli(double dollary)
    {
        double result = dollary * kurs;
        Console.WriteLine($"{dollary}$ = {result} руб.");
        return result;
    }

    public static double RubliVDollary(double rubli)
    {
        double result = rubli / kurs;
        Console.WriteLine($"{rubli} руб. = {result}$");
        return result;
    }
}

public class Krug
{
    public const double PI = 3.14159;
    public double Radius;

    public Krug(double r)
    {
        Radius = r;
    }

    public double Ploshad()
    {
        return PI * Radius * Radius;
    }

    public void Pokazat()
    {
        Console.WriteLine($"Радиус: {Radius}, Площадь: {Ploshad():F2}");
    }
}

class Program
{
    static void Main()
    {
        Console.OutputEncoding = System.Text.Encoding.UTF8;

        Console.WriteLine("ЗАДАЧА 3: Конвертер валют\n");

        Console.Write("Долларов в рубли: ");
        double d = double.Parse(Console.ReadLine());
        CurrencyConverter.DollaryVRubli(d);

        Console.Write("Рублей в доллары: ");
        double r = double.Parse(Console.ReadLine());
        CurrencyConverter.RubliVDollary(r);

        Console.Write("Новый курс: ");
        double k = double.Parse(Console.ReadLine());
        CurrencyConverter.SmenitKurs(k);

        Console.Write("Еще долларов: ");
        d = double.Parse(Console.ReadLine());
        CurrencyConverter.DollaryVRubli(d);

        Console.WriteLine("\nЗАДАЧА 4: Круг\n");

        Console.Write("Радиус 1: ");
        Krug k1 = new Krug(double.Parse(Console.ReadLine()));

        Console.Write("Радиус 2: ");
        Krug k2 = new Krug(double.Parse(Console.ReadLine()));

        Console.WriteLine("\nКруг 1:");
        k1.Pokazat();

        Console.WriteLine("\nКруг 2:");
        k2.Pokazat();

        Console.ReadKey();
    }
}
