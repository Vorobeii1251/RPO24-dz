import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Profile Card with Status',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const ProfileScreen(),
    );
  }
}

class ProfileScreen extends StatelessWidget {
  const ProfileScreen({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text('Профиль со статусом'),
        centerTitle: true,
        backgroundColor: Colors.blue,
        foregroundColor: Colors.white,
      ),
      body: const Center(
        child: ProfileCard(),
      ),
    );
  }
}

class ProfileCard extends StatefulWidget {
  const ProfileCard({super.key});

  @override
  State<ProfileCard> createState() => _ProfileCardState();
}

class _ProfileCardState extends State<ProfileCard> {
  final List<String> _statuses = [
    'Отдыхаю',
    'Работаю', 
    'В отпуске',
    'Не в сети'
  ];
  
  final List<Color> _statusColors = [
    Colors.green,     // Отдыхаю - зеленый
    Colors.blue,      // Работаю - голубой
    Colors.orange,    // В отпуске - оранжевый
    Colors.grey,      // Не в сети - серый
  ];
  
  int _currentStatusIndex = 0;
  
  Color _cardColor = Colors.green.shade50;
  
  @override
  void initState() {
    super.initState();
    _updateCardColor();
  }
  
  void _updateCardColor() {
    setState(() {
      _cardColor = _statusColors[_currentStatusIndex].withOpacity(0.1);
    });
  }
  
  void _changeStatus() {
    setState(() {
      _currentStatusIndex = (_currentStatusIndex + 1) % _statuses.length;
      _cardColor = _statusColors[_currentStatusIndex].withOpacity(0.1);
    });
  }
  
  void _changeBackgroundColor(Color color) {
    ScaffoldMessenger.of(context).showSnackBar(
      SnackBar(
        content: Text('Цвет фона изменен на ${_getColorName(color)}'),
        backgroundColor: color,
        duration: const Duration(seconds: 1),
      ),
    );
  }
  
  String _getColorName(Color color) {
    if (color == Colors.red) return 'красный';
    if (color == Colors.green) return 'зеленый';
    if (color == Colors.blue) return 'синий';
    if (color == Colors.orange) return 'оранжевый';
    if (color == Colors.purple) return 'фиолетовый';
    return 'другой';
  }

  @override
  Widget build(BuildContext context) {
    return Column(
      mainAxisAlignment: MainAxisAlignment.center,
      children: [
        Container(
          width: 350,
          decoration: BoxDecoration(
            color: _cardColor,
            borderRadius: BorderRadius.circular(20),
            boxShadow: [
              BoxShadow(
                color: Colors.grey.withOpacity(0.3),
                blurRadius: 10,
                offset: const Offset(0, 4),
              ),
            ],
            border: Border.all(
              color: _statusColors[_currentStatusIndex].withOpacity(0.3),
              width: 2,
            ),
          ),
          child: Column(
            children: [
              const Padding(
                padding: EdgeInsets.all(20),
                child: Column(
                  children: [
                    CircleAvatar(
                      radius: 50,
                      backgroundColor: Colors.blue,
                      child: Icon(
                        Icons.person,
                        size: 50,
                        color: Colors.white,
                      ),
                    ),
                    SizedBox(height: 16),
                    Text(
                      'Nikita pypsik',
                      style: TextStyle(
                        fontSize: 24,
                        fontWeight: FontWeight.bold,
                      ),
                    ),
                    SizedBox(height: 8),
                    Text(
                      'лучший программист в мире',
                      style: TextStyle(
                        fontSize: 16,
                        color: Colors.grey,
                      ),
                    ),
                  ],
                ),
              ),
              
              Container(
                margin: const EdgeInsets.symmetric(horizontal: 20),
                padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 12),
                decoration: BoxDecoration(
                  color: Colors.white,
                  borderRadius: BorderRadius.circular(12),
                  border: Border.all(color: Colors.grey.shade300),
                ),
                child: Row(
                  children: [
                    Container(
                      padding: const EdgeInsets.all(8),
                      decoration: BoxDecoration(
                        color: _statusColors[_currentStatusIndex].withOpacity(0.1),
                        borderRadius: BorderRadius.circular(8),
                      ),
                      child: Icon(
                        Icons.info_outline,
                        color: _statusColors[_currentStatusIndex],
                        size: 24,
                      ),
                    ),
                    const SizedBox(width: 16),
                    const Expanded(
                      child: Column(
                        crossAxisAlignment: CrossAxisAlignment.start,
                        children: [
                          Text(
                            'Статус',
                            style: TextStyle(
                              fontSize: 12,
                              color: Colors.grey,
                            ),
                          ),
                          SizedBox(height: 4),
                          Text(
                            'Текущий статус',
                            style: TextStyle(
                              fontSize: 16,
                              fontWeight: FontWeight.w500,
                            ),
                          ),
                        ],
                      ),
                    ),
                    Container(
                      padding: const EdgeInsets.symmetric(horizontal: 12, vertical: 6),
                      decoration: BoxDecoration(
                        color: _statusColors[_currentStatusIndex].withOpacity(0.1),
                        borderRadius: BorderRadius.circular(20),
                      ),
                      child: Text(
                        _statuses[_currentStatusIndex],
                        style: TextStyle(
                          color: _statusColors[_currentStatusIndex],
                          fontWeight: FontWeight.w600,
                        ),
                      ),
                    ),
                  ],
                ),
              ),
              
              const SizedBox(height: 16),
              
              Padding(
                padding: const EdgeInsets.symmetric(horizontal: 20),
                child: ElevatedButton(
                  onPressed: _changeStatus,
                  style: ElevatedButton.styleFrom(
                    backgroundColor: _statusColors[_currentStatusIndex],
                    foregroundColor: Colors.white,
                    minimumSize: const Size(double.infinity, 50),
                    shape: RoundedRectangleBorder(
                      borderRadius: BorderRadius.circular(12),
                    ),
                  ),
                  child: const Text(
                    'Сменить статус',
                    style: TextStyle(fontSize: 16),
                  ),
                ),
              ),
              
              const SizedBox(height: 20),
            ],
          ),
        ),
        
        const SizedBox(height: 30),
        
        const Text(
          'Изменить цвет фона:',
          style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
        ),
        
        const SizedBox(height: 16),
        
        Row(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            _buildColorButton(Colors.red, 'Красный'),
            const SizedBox(width: 10),
            _buildColorButton(Colors.green, 'Зеленый'),
            const SizedBox(width: 10),
            _buildColorButton(Colors.blue, 'Синий'),
          ],
        ),
        
        const SizedBox(height: 10),
        
        Row(
          mainAxisAlignment: MainAxisAlignment.center,
          children: [
            _buildColorButton(Colors.orange, 'Оранжевый'),
            const SizedBox(width: 10),
            _buildColorButton(Colors.purple, 'Фиолетовый'),
            const SizedBox(width: 10),
            _buildColorButton(Colors.grey, 'Серый'),
          ],
        ),
      ],
    );
  }
  
  Widget _buildColorButton(Color color, String label) {
    return ElevatedButton(
      onPressed: () => _changeBackgroundColor(color),
      style: ElevatedButton.styleFrom(
        backgroundColor: color,
        foregroundColor: Colors.white,
        minimumSize: const Size(90, 40),
        shape: RoundedRectangleBorder(
          borderRadius: BorderRadius.circular(20),
        ),
      ),
      child: Text(label),
    );
  }
}
